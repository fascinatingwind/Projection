\begin{bmatrix}
yB - yA & xA - xB & 0 \\
zB - zA & 0 & xA - xB \\
0 & zB - zA & yB - yA \\
\end{bmatrix}

\begin{bmatrix}
yB - yA & xA - xB & 0 \\
zB - zA & 0 & xA - xB \\
xB - xA & yB - yA & zB - zA \\
\end{bmatrix}

\begin{bmatrix}
xA(yB - yA) - yA(xB - xA)\\
yA(zB - zA) - zA(xB - xA)\\
xC(xB - xA) + yC(yB - yA) + zC(zB - zA)\\
\end{bmatrix}
